def run_orca(f):
    cmd = '/home/shad/progs/bin/orca ' + f
    p = subprocess.Popen(cmd, shell=True, stdout=subprocess.PIPE, stderr=subprocess.PIPE)
    out = p.communicate()[0]
    lines = out.splitlines()
    res = 'inf'
    for l in lines:
        if 'FINAL SINGLE POINT ENERGY' in l:
            res = l.split()[4]
    return float(res)

def dep_plot(x,y,x_lim1,x_lim2):
    x_o = x
    y_o = y
    # function is  f(x)=k(b-x)^2 + a
    fitfunc = lambda p, x: p[0]*pow(p[1]-x,2) + p[2] # Target function
    errfunc = lambda p, x, y: fitfunc(p, x) - y # Error function
    p0 = [1,1, -79] # Initial guess for the parameters
    
    p1, success = optimize.leastsq(errfunc, p0[:], args=(x_o, y_o))
    print 'Optimized params:', p1
    
    plt.plot(x_o, y_o, 'ro', x_o,fitfunc(p1,x_o),'r-',c='blue',alpha=0.5)
    plt.xlim(x_lim1,x_lim2)
    plt.show()

Сравнение полученных констант с данными из статьи о GAFF:¶

Из статьи:¶

В статье вычисления делаются на основе методов молекулярной динамики, а расчёты в ORCA - методами квантовой химии.

Эмпирическое уравнение для связей С-С

$ ln(Kr)=−4.9575⋅r+7.8187 $

Оптимальная длиная связи (по статье):

$ r=1.526 $

$ K_{r} = exp(−4.9526 * 1.526 + 7.8187) = 306.582Ккал/моль $

Тут :

$ K_{r} = 0.66824639Eh * 627,51Ккал/моль = 419.331Ккал/моль $

В статье о GAFF говориться, что k ~ 50 ккал/мольxрад^2, величина b явно не указана явно.
